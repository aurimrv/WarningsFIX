28/**
29 * A configurable {@link ObjectPool} implementation.
30 * <p>
31 * When coupled with the appropriate {@link PoolableObjectFactory},
32 * <tt>GenericObjectPool</tt> provides robust pooling functionality for
33 * arbitrary objects.
34 * <p>
35 * A <tt>GenericObjectPool</tt> provides a number of configurable parameters:
36 * <ul>
37 *  <li>
38 *    {@link #setMaxActive <i>maxActive</i>} controls the maximum number of
39 *    objects that can be borrowed from the pool at one time.  When
40 *    non-positive, there is no limit to the number of objects that may be
41 *    active at one time. When {@link #setMaxActive <i>maxActive</i>} is
42 *    exceeded, the pool is said to be exhausted. The default setting for this
43 *    parameter is 8.
44 *  </li>
45 *  <li>
46 *    {@link #setMaxIdle <i>maxIdle</i>} controls the maximum number of objects
47 *    that can sit idle in the pool at any time.  When negative, there is no
48 *    limit to the number of objects that may be idle at one time. The default
49 *    setting for this parameter is 8.
50 *  </li>
51 *  <li>
52 *    {@link #setWhenExhaustedAction <i>whenExhaustedAction</i>} specifies the
53 *    behavior of the {@link #borrowObject} method when the pool is exhausted:
54 *    <ul>
55 *    <li>
56 *      When {@link #setWhenExhaustedAction <i>whenExhaustedAction</i>} is
57 *      {@link #WHEN_EXHAUSTED_FAIL}, {@link #borrowObject} will throw
58 *      a {@link NoSuchElementException}
59 *    </li>
60 *    <li>
61 *      When {@link #setWhenExhaustedAction <i>whenExhaustedAction</i>} is
62 *      {@link #WHEN_EXHAUSTED_GROW}, {@link #borrowObject} will create a new
63 *      object and return it(essentially making {@link #setMaxActive <i>maxActive</i>}
64 *      meaningless.)
65 *    </li>
66 *    <li>
67 *      When {@link #setWhenExhaustedAction <i>whenExhaustedAction</i>}
68 *      is {@link #WHEN_EXHAUSTED_BLOCK}, {@link #borrowObject} will block
69 *      (invoke {@link Object#wait()} until a new or idle object is available.
70 *      If a positive {@link #setMaxWait <i>maxWait</i>}
71 *      value is supplied, the {@link #borrowObject} will block for at
72 *      most that many milliseconds, after which a {@link NoSuchElementException}
73 *      will be thrown.  If {@link #setMaxWait <i>maxWait</i>} is non-positive,
74 *      the {@link #borrowObject} method will block indefinitely.
75 *    </li>
76 *    </ul>
77 *    The default <code>whenExhaustedAction</code> setting is
78 *    {@link #WHEN_EXHAUSTED_BLOCK} and the default <code>maxWait</code>
79 *    setting is -1. By default, therefore, <code>borrowObject</code> will
80 *    block indefinitely until an idle instance becomes available.
81 *  </li>
82 *  <li>
83 *    When {@link #setTestOnBorrow <i>testOnBorrow</i>} is set, the pool will
84 *    attempt to validate each object before it is returned from the
85 *    {@link #borrowObject} method. (Using the provided factory's
86 *    {@link PoolableObjectFactory#validateObject} method.)  Objects that fail
87 *    to validate will be dropped from the pool, and a different object will
88 *    be borrowed. The default setting for this parameter is
89 *    <code>false.</code>
90 *  </li>
91 *  <li>
92 *    When {@link #setTestOnReturn <i>testOnReturn</i>} is set, the pool will
93 *    attempt to validate each object before it is returned to the pool in the
94 *    {@link #returnObject} method. (Using the provided factory's
95 *    {@link PoolableObjectFactory#validateObject}
96 *    method.)  Objects that fail to validate will be dropped from the pool.
97 *    The default setting for this parameter is <code>false.</code>
98 *  </li>
99 * </ul>
100 * <p>
101 * Optionally, one may configure the pool to examine and possibly evict objects
102 * as they sit idle in the pool and to ensure that a minimum number of idle
103 * objects are available. This is performed by an "idle object eviction"
104 * thread, which runs asynchronously. Caution should be used when configuring
105 * this optional feature. Eviction runs require an exclusive synchronization
106 * lock on the pool, so if they run too frequently and / or incur excessive
107 * latency when creating, destroying or validating object instances,
108 * performance issues may result.  The idle object eviction thread may be
109 * configured using the following attributes:
110 * <ul>
111 *  <li>
112 *   {@link #setTimeBetweenEvictionRunsMillis <i>timeBetweenEvictionRunsMillis</i>}
113 *   indicates how long the eviction thread should sleep before "runs" of examining
114 *   idle objects.  When non-positive, no eviction thread will be launched. The
115 *   default setting for this parameter is -1 (i.e., idle object eviction is
116 *   disabled by default).
117 *  </li>
118 *  <li>
119 *   {@link #setMinEvictableIdleTimeMillis <i>minEvictableIdleTimeMillis</i>}
120 *   specifies the minimum amount of time that an object may sit idle in the pool
121 *   before it is eligible for eviction due to idle time.  When non-positive, no object
122 *   will be dropped from the pool due to idle time alone. This setting has no
123 *   effect unless <code>timeBetweenEvictionRunsMillis > 0.</code> The default
124 *   setting for this parameter is 30 minutes.
125 *  </li>
126 *  <li>
127 *   {@link #setTestWhileIdle <i>testWhileIdle</i>} indicates whether or not idle
128 *   objects should be validated using the factory's
129 *   {@link PoolableObjectFactory#validateObject} method. Objects that fail to
130 *   validate will be dropped from the pool. This setting has no effect unless
131 *   <code>timeBetweenEvictionRunsMillis > 0.</code>  The default setting for
132 *   this parameter is <code>false.</code>
133 *  </li>
134 *  <li>
135 *   {@link #setSoftMinEvictableIdleTimeMillis <i>softMinEvictableIdleTimeMillis</i>}
136 *   specifies the minimum amount of time an object may sit idle in the pool
137 *   before it is eligible for eviction by the idle object evictor
138 *   (if any), with the extra condition that at least "minIdle" amount of object
139 *   remain in the pool.  When non-positive, no objects will be evicted from the pool
140 *   due to idle time alone. This setting has no effect unless
141 *   <code>timeBetweenEvictionRunsMillis > 0.</code>  The default setting for
142 *   this parameter is -1 (disabled).
143 *  </li>
144 *  <li>
145 *   {@link #setNumTestsPerEvictionRun <i>numTestsPerEvictionRun</i>}
146 *   determines the number of objects examined in each run of the idle object
147 *   evictor. This setting has no effect unless
148 *   <code>timeBetweenEvictionRunsMillis > 0.</code>  The default setting for
149 *   this parameter is 3.
150 *  </li>
151 * </ul>
152 * <p>
153 * <p>
154 * The pool can be configured to behave as a LIFO queue with respect to idle
155 * objects - always returning the most recently used object from the pool,
156 * or as a FIFO queue, where borrowObject always returns the oldest object
157 * in the idle object pool.
158 * <ul>
159 *  <li>
160 *   {@link #setLifo <i>lifo</i>}
161 *   determines whether or not the pool returns idle objects in
162 *   last-in-first-out order. The default setting for this parameter is
163 *   <code>true.</code>
164 *  </li>
165 * </ul>
166 * <p>
167 * GenericObjectPool is not usable without a {@link PoolableObjectFactory}.  A
168 * non-<code>null</code> factory must be provided either as a constructor argument

910    public Object borrowObject() throws Exception {
911        long starttime = System.currentTimeMillis();
912        for(;;) {
913            ObjectTimestampPair pair = null;
914
915            synchronized (this) {
916                assertOpen();
917                // if there are any sleeping, just grab one of those
918                try {
919                    pair = (ObjectTimestampPair)(_pool.removeFirst());
920                } catch(NoSuchElementException e) {
921                    ; /* ignored */
922                }
923
924                // otherwise
925                if(null == pair) {
926                    // check if we can create one
927                    // (note we know that the num sleeping is 0, else we wouldn't be here)
928                    if(_maxActive < 0 || _numActive < _maxActive) {
929                        // allow new object to be created
930                    } else {
931                        // the pool is exhausted
932                        switch(_whenExhaustedAction) {
933                        case WHEN_EXHAUSTED_GROW:
934                            // allow new object to be created
935                            break;
936                        case WHEN_EXHAUSTED_FAIL:
937                            throw new NoSuchElementException("Pool exhausted");
938                        case WHEN_EXHAUSTED_BLOCK:
939                            try {
940                                if(_maxWait <= 0) {
941                                    wait();
942                                } else {
943                                    // this code may be executed again after a notify then continue cycle
944                                    // so, need to calculate the amount of time to wait
945                                    final long elapsed = (System.currentTimeMillis() - starttime);
946                                    final long waitTime = _maxWait - elapsed;
947                                    if (waitTime > 0) {
948                                        wait(waitTime);
949                                    }
950                                }
951                            } catch(InterruptedException e) {
952                                Thread.currentThread().interrupt();
953                                throw e;
954                            }
955                            if(_maxWait > 0 && ((System.currentTimeMillis() - starttime) >= _maxWait)) {
956                                throw new NoSuchElementException("Timeout waiting for idle object");
957                            } else {
958                                continue; // keep looping
959                            }
960                        default:
961                            throw new IllegalArgumentException("WhenExhaustedAction property " + _whenExhaustedAction + " not recognized.");
962                        }
963                    }
964                }
965                _numActive++;
966            }
967
968            // create new object when needed
969            boolean newlyCreated = false;
970            if(null == pair) {
971                try {
972                    Object obj = _factory.makeObject();
973                    pair = new ObjectTimestampPair(obj);
974                    newlyCreated = true;
975                } finally {
976                    if (!newlyCreated) {
977                        // object cannot be created
978                        synchronized (this) {
979                            _numActive--;
980                            notifyAll();
981                        }
982                    }
983                }
984            }
985
986            // activate & validate the object
987            try {
988                _factory.activateObject(pair.value);
989                if(_testOnBorrow && !_factory.validateObject(pair.value)) {
990                    throw new Exception("ValidateObject failed");
991                }
992                return pair.value;
993            } catch (Throwable e) {
994                // object cannot be activated or is invalid
995                try {
996                    _factory.destroyObject(pair.value);
997                } catch (Throwable e2) {
998                    // cannot destroy broken object
999                }
1000                synchronized (this) {
1001                    _numActive--;
1002                    notifyAll();
1003                }
1004                if(newlyCreated) {
1005                    throw new NoSuchElementException("Could not create a validated object, cause: " + e.getMessage());
1006                } else {
1007                    continue; // keep looping
1008                }
1009            }
1010        }
1011    }

1014        try {
1015            if (_factory != null) {
1016                _factory.destroyObject(obj);
1017            }
1018        } finally {
1019            synchronized (this) {
1020                _numActive--;
1021                notifyAll(); // _numActive has changed
1022            }
1023        }

1060    /**
1061     * {@inheritDoc}
1062     * <p><strong>Note: </strong> There is no guard to prevent an object
1063     * being returned to the pool multiple times. Clients are expected to
1064     * discard references to returned objects and ensure that an object is not
1065     * returned to the pool multiple times in sequence (i.e., without being
1066     * borrowed again between returns). Violating this contract will result in
1067     * the same object appearing multiple times in the pool and pool counters
1068     * (numActive, numIdle) returning incorrect values.</p>
1069     */
1070    public void returnObject(Object obj) throws Exception {
1071        try {
1072            addObjectToPool(obj, true);
1073        } catch (Exception e) {
1074            if (_factory != null) {
1075                try {
1076                    _factory.destroyObject(obj);
1077                } catch (Exception e2) {
1078                    // swallowed
1079                }
1080                // TODO: Correctness here depends on control in addObjectToPool.
1081                // These two methods should be refactored, removing the
1082                // "behavior flag",decrementNumActive, from addObjectToPool.
1083                synchronized(this) {
1084                    _numActive--;
1085                    notifyAll();
1086                }
1087            }
1088        }
1089    }

1091    private void addObjectToPool(Object obj, boolean decrementNumActive) throws Exception {
1092        boolean success = true;
1093        if(_testOnReturn && !(_factory.validateObject(obj))) {
1094            success = false;
1095        } else {
1096            _factory.passivateObject(obj);
1097        }
1098
1099        boolean shouldDestroy = !success;
1100
1101        // Add instance to pool if there is room and it has passed validation
1102        // (if testOnreturn is set)
1103        synchronized (this) {
1104            if (isClosed()) {
1105                shouldDestroy = true;
1106            } else {
1107                if((_maxIdle >= 0) && (_pool.size() >= _maxIdle)) {
1108                    shouldDestroy = true;
1109                } else if(success) {
1110                    // borrowObject always takes the first element from the queue,
1111                    // so for LIFO, push on top, FIFO add to end
1112                    if (_lifo) {
1113                        _pool.addFirst(new ObjectTimestampPair(obj));
1114                    } else {
1115                        _pool.addLast(new ObjectTimestampPair(obj));
1116                    }
1117                }
1118            }
1119        }
1120
1121        // Destroy the instance if necessary
1122        if(shouldDestroy) {
1123            try {
1124                _factory.destroyObject(obj);
1125            } catch(Exception e) {
1126                // ignored
1127            }
1128        }
1129
1130        // Decrement active count *after* destroy if applicable
1131        if (decrementNumActive) {
1132            synchronized(this) {
1133                _numActive--;
1134                notifyAll();
1135            }
1136        }
1137    }

1166    /**
1167     * <p>Perform <code>numTests</code> idle object eviction tests, evicting
1168     * examined objects that meet the criteria for eviction. If
1169     * <code>testWhileIdle</code> is true, examined objects are validated
1170     * when visited (and removed if invalid); otherwise only objects that
1171     * have been idle for more than <code>minEvicableIdletimeMillis</code>
1172     * are removed.</p>
1173     *
1174     * <p>Successive activations of this method examine objects in
1175     * in sequence, cycling through objects in oldest-to-youngest order.</p>
1176     *
1177     * @throws Exception if the pool is closed or eviction fails.
1178     */
1179    public synchronized void evict() throws Exception {
1180        assertOpen();
1181        if(!_pool.isEmpty()) {
1182            if (null == _evictionCursor) {
1183                _evictionCursor = (_pool.cursor(_lifo ? _pool.size() : 0));
1184            }
1185            for (int i=0,m=getNumTests(); i<m; i++) {
1186                if ((_lifo && !_evictionCursor.hasPrevious()) ||
1187                        !_lifo && !_evictionCursor.hasNext()) {
1188                    _evictionCursor.close();
1189                    _evictionCursor = _pool.cursor(_lifo ? _pool.size() : 0);
1190                }
1191                boolean removeObject = false;
1192                final ObjectTimestampPair pair = _lifo ?
1193                                                 (ObjectTimestampPair) _evictionCursor.previous() :
1194                                                 (ObjectTimestampPair) _evictionCursor.next();
1195                final long idleTimeMilis = System.currentTimeMillis() - pair.tstamp;
1196                if ((_minEvictableIdleTimeMillis > 0)
1197                        && (idleTimeMilis > _minEvictableIdleTimeMillis)) {
1198                    removeObject = true;
1199                } else if ((_softMinEvictableIdleTimeMillis > 0)
1200                           && (idleTimeMilis > _softMinEvictableIdleTimeMillis)
1201                           && (getNumIdle() > getMinIdle())) {
1202                    removeObject = true;
1203                }
1204                if(_testWhileIdle && !removeObject) {
1205                    boolean active = false;
1206                    try {
1207                        _factory.activateObject(pair.value);
1208                        active = true;
1209                    } catch(Exception e) {
1210                        removeObject=true;
1211                    }
1212                    if(active) {
1213                        if(!_factory.validateObject(pair.value)) {
1214                            removeObject=true;
1215                        } else {
1216                            try {
1217                                _factory.passivateObject(pair.value);
1218                            } catch(Exception e) {
1219                                removeObject=true;
1220                            }
1221                        }
1222                    }
1223                }
1224                if(removeObject) {
1225                    try {
1226                        _evictionCursor.remove();
1227                        _factory.destroyObject(pair.value);
1228                    } catch(Exception e) {
1229                        // ignored
1230                    }
1231                }
1232            }
1233        } // if !empty
1234    }

1236    /**
1237     * Check to see if we are below our minimum number of objects
1238     * if so enough to bring us back to our minimum.
1239     *
1240     * @throws Exception when {@link #addObject()} fails.
1241     */
1242    protected void ensureMinIdle() throws Exception {
1243        // this method isn't synchronized so the
1244        // calculateDeficit is done at the beginning
1245        // as a loop limit and a second time inside the loop
1246        // to stop when another thread already returned the
1247        // needed objects
1248        int objectDeficit = calculateDeficit();
1249        for ( int j = 0 ; j < objectDeficit && calculateDeficit() > 0 ; j++ ) {
1250            addObject();
1251        }
1252    }
1253
1254    private synchronized int calculateDeficit() {
1255        int objectDeficit = getMinIdle() - getNumIdle();
1256        if (_maxActive > 0) {
1257            int growLimit = Math.max(0, getMaxActive() - getNumActive() - getNumIdle());
1258            objectDeficit = Math.min(objectDeficit, growLimit);
1259        }
1260        return objectDeficit;
1261    }

1263    /**
1264     * Create an object, and place it into the pool.
1265     * addObject() is useful for "pre-loading" a pool with idle objects.
1266     */
1267    public synchronized void addObject() throws Exception {
1268        assertOpen();
1269        if (_factory == null) {
1270            throw new IllegalStateException("Cannot add objects without a factory.");
1271        }
1272        Object obj = _factory.makeObject();
1273        try {
1274            assertOpen();
1275            addObjectToPool(obj, false);
1276        } catch (IllegalStateException ex) { // Pool closed
1277            try {
1278                _factory.destroyObject(obj);
1279            } catch (Exception ex2) {
1280                // swallow
1281            }
1282            throw ex;
1283        }
1284    }

1350     * A simple "struct" encapsulating an object instance and a timestamp.
1351     *
1352     * Implements Comparable, objects are sorted from old to new.
1353     *
1354     * This is also used by {@link GenericObjectPool}.
1355     */
1356    static class ObjectTimestampPair implements Comparable {
1357        Object value;
1358        long tstamp;
1359
1360        ObjectTimestampPair(Object val) {
1361            this(val, System.currentTimeMillis());
1362        }
1363
1364        ObjectTimestampPair(Object val, long time) {
1365            value = val;
1366            tstamp = time;
1367        }
1368
1369        public String toString() {
1370            return value + ";" + tstamp;
1371        }
1372
1373        public int compareTo(Object obj) {
1374            return compareTo((ObjectTimestampPair) obj);
1375        }
1376
1377        public int compareTo(ObjectTimestampPair other) {
1378            final long tstampdiff = this.tstamp - other.tstamp;
1379            if (tstampdiff == 0) {
1380                // make sure the natural ordering is consistent with equals
1381                // see java.lang.Comparable Javadocs
1382                return System.identityHashCode(this) - System.identityHashCode(other);
1383            } else {
1384                // handle int overflow
1385                return (int)Math.min(Math.max(tstampdiff, Integer.MIN_VALUE), Integer.MAX_VALUE);
1386            }
1387        }
1388    }
